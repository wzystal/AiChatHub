## SseClient.kt（核心实现骨架）

👉 这里是你项目“技术含金量核心”

```kotlin
class SseClient {

    fun startStream(
        prompt: String,
        onToken: (String) -> Unit,
        onComplete: () -> Unit,
        onError: (Throwable) -> Unit
    ) {
        val request = Request.Builder()
            .url("YOUR_LLM_API_STREAM_URL")
            .build()

        val client = OkHttpClient()

        client.newCall(request).enqueue(object : Callback {

            override fun onFailure(call: Call, e: IOException) {
                onError(e)
            }

            override fun onResponse(call: Call, response: Response) {
                val source = response.body?.source() ?: return

                try {
                    while (!source.exhausted()) {
                        val line = source.readUtf8Line() ?: continue

                        // 假设每行是一个 token
                        if (line.startsWith("data:")) {
                            val token = line.removePrefix("data:")
                            onToken(token)
                        }
                    }

                    onComplete()

                } catch (e: Exception) {
                    onError(e)
                }
            }
        })
    }
}
```

## ChatRepository.kt

```kotlin
class ChatRepository(
    private val sseClient: SseClient
) {

    fun sendMessageStream(
        messages: List<ChatMessage>,
        onToken: (String) -> Unit,
        onComplete: () -> Unit,
        onError: (Throwable) -> Unit
    ) {
        val prompt = messages.last().content

        sseClient.startStream(
            prompt = prompt,
            onToken = onToken,
            onComplete = onComplete,
            onError = onError
        )
    }
}
```

# 🧠 6. ViewModel（核心逻辑）

## ChatViewModel.kt

👉 这里是“流式 + UI更新核心”

```kotlin
class ChatViewModel(
    private val repo: ChatRepository
) : ViewModel() {

    private val _uiState = MutableStateFlow(ChatUiState())
    val uiState = _uiState.asStateFlow()

    fun sendMessage(input: String) {

        val userMsg = ChatMessage(
            id = System.currentTimeMillis().toString(),
            role = Role.USER,
            content = input
        )

        val assistantMsg = ChatMessage(
            id = "ai_${System.currentTimeMillis()}",
            role = Role.ASSISTANT,
            content = "",
            isStreaming = true
        )

        val newList = _uiState.value.messages + userMsg + assistantMsg
        _uiState.value = _uiState.value.copy(messages = newList)

        streamAIResponse(newList, assistantMsg.id)
    }

    private fun streamAIResponse(
        messages: List<ChatMessage>,
        assistantId: String
    ) {
        repo.sendMessageStream(
            messages = messages,
            onToken = { token ->
                updateLastAssistantMessage(assistantId, token)
            },
            onComplete = {
                finishStreaming(assistantId)
            },
            onError = {
                // TODO error handle
            }
        )
    }

    private fun updateLastAssistantMessage(id: String, token: String) {
        val updated = _uiState.value.messages.map {
            if (it.id == id) {
                it.copy(content = it.content + token)
            } else it
        }

        _uiState.value = _uiState.value.copy(messages = updated)
    }

    private fun finishStreaming(id: String) {
        val updated = _uiState.value.messages.map {
            if (it.id == id) {
                it.copy(isStreaming = false)
            } else it
        }

        _uiState.value = _uiState.value.copy(messages = updated)
    }
}
```

# 📱 7. UI层（Activity）

## ChatActivity.kt

```kotlin
class ChatActivity : AppCompatActivity() {

    private lateinit var adapter: ChatAdapter
    private val viewModel: ChatViewModel by viewModels()

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)

        setContentView(R.layout.activity_chat)

        setupRecyclerView()

        lifecycleScope.launch {
            viewModel.uiState.collect { state ->
                adapter.submitList(state.messages)
            }
        }

        findViewById<Button>(R.id.sendBtn).setOnClickListener {
            val input = findViewById<EditText>(R.id.input).text.toString()
            viewModel.sendMessage(input)
        }
    }

    private fun setupRecyclerView() {
        adapter = ChatAdapter()
        findViewById<RecyclerView>(R.id.recyclerView).adapter = adapter
    }
}
```

# ⚡ 8. RecyclerView优化核心（重点）

## ChatAdapter.kt（payload版本）

```kotlin
class ChatAdapter :
    ListAdapter<ChatMessage, ChatViewHolder>(DIFF) {

    override fun onBindViewHolder(
        holder: ChatViewHolder,
        position: Int,
        payloads: MutableList<Any>
    ) {
        if (payloads.isNotEmpty()) {
            holder.updateContent(payloads[0] as String)
        } else {
            super.onBindViewHolder(holder, position, payloads)
        }
    }

    override fun onBindViewHolder(holder: ChatViewHolder, position: Int) {
        holder.bind(getItem(position))
    }

    companion object {
        val DIFF = object : DiffUtil.ItemCallback<ChatMessage>() {
            override fun areItemsTheSame(a: ChatMessage, b: ChatMessage) =
                a.id == b.id

            override fun areContentsTheSame(a: ChatMessage, b: ChatMessage) =
                a.content == b.content

            override fun getChangePayload(oldItem: ChatMessage, newItem: ChatMessage): Any? {
                return newItem.content
            }
        }
    }
}
```

## ChatViewHolder.kt

```kotlin
class ChatViewHolder(itemView: View) : RecyclerView.ViewHolder(itemView) {

    fun bind(msg: ChatMessage) {
        itemView.findViewById<TextView>(R.id.text).text = msg.content
    }

    fun updateContent(content: String) {
        itemView.findViewById<TextView>(R.id.text).text = content
    }
}
```
